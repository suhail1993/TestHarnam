CREATE TABLE student_register  (
    std_id int NOT NULL AUTO_INCREMENT,
    Name varchar(30) NOT NULL,
    city varchar(25),
    dob varchar(25),
	street1 varchar(50,
	street2 varchar(30),
	street3 varchar(30,
	contact varchar(30),
	state varchar(30,
	pin varchar(20),
	country varchar(35),
	email varchar(30),
	password varchar(30),
	marks varchar(20),
    PRIMARY KEY (std_id)
);
